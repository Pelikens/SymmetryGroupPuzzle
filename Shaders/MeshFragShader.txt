// MeshFragShader.txt

precision mediump float;

varying vec2 frag_texCoord;

uniform sampler2D texSampler;
uniform vec4 blend_factor;
uniform vec4 const_color;

void main()
{
    vec4 tex_color = texture2D(texSampler, vec2(frag_texCoord.s, frag_texCoord.y));

    gl_FragColor = blend_factor * tex_color + (vec4(1.0, 1.0, 1.0, 1.0) - blend_factor) * const_color;
}
